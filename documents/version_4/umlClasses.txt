class Personnage{
- posX : int
- posY : int 
- pv : int 
- vieMax : int 
- mort : boolean
+ Personnage(int, int) : void
+ deplacer(int, int) : void
+ gagnerVie(int) : void
+ prendreDegats(int) : void
+ getX() : int
+ getY() : int
+ getPv() : int 
+ getVieMax() : int

}

abstract class Case{
- x : int 
- y : int 
- franchissable : boolean
+ Case(int, int, boolean)
+ getX() : int 
+ getY() : int
+ estFranchissable() : boolean
}

class Labyrinthe{
- xLong : int
- yLong : int
+ Labyrinthe(String)
+ estAccessible(int, int) : boolean
+ estUnPiege(int, int) : boolean
+ getTailleX() : int
+ getTailleY() : int
+ getTableauCases() : Case[][]
}

}

+ Sol(int, int)
}

class Sol {

}

class Piege{
+ Piege(int, int)

abstract class Monstre <<Nouveau>>{
- posX : int
- posY : int
- vie : int
- ptsAttaque : int
+ Monstre(int, int, int, int)
+ attaquer(Personnage) : void
+ subirDegats(int) : void
+ deplacer(int, int) : void
+ getPosX() : int
+ getPosY() : int
+ getVie() : int

}

class Troll <<Nouveau>>{
+ Troll(int, int)
}

interface Entite <<Nouveau>>{
+ attaquer(Entite) : void
+ prendreDegats(int) : void
+ deplacer(int, int) : void
}

Labyrinthe "0" -> "*" Case : -cases

Mur --|> Case
Sol --|> Case
Piege --|> Case

Troll --|> Monstre

Personnage ..|> Entite
Monstre ..|> Entite

DessinZ "*" -> "1" JeuZ : -jeu

JeuZ "*" -left> "1" Personnage : -aventurier
JeuZ "*" -> "1" Labyrinthe : -niveaux
